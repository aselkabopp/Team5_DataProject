create or replace view "TARGITO_207F62E1-FC2D-4D55-9C5C-B800034DCC62".ANALYSIS.MAILING_ACTION(
	MAILING_ID,
	MAILING_DATE,
	CONTACT_ID,
	CREATED_DATE,
	ORIGIN,
	NAME,
	GENDER_NEW,
	SOURCE
) as 
select m.MAILING_ID,
m.MAILING_DATE,
m.CONTACT_ID,
m.CREATED_DATE,
m.ORIGIN,
m.NAME,
CASE 
    WHEN c.gender_new is null
    THEN 'unknown'
    ELSE c.gender_new
END gender_new,
CASE 
    WHEN c.source is null
    THEN 'unknown'
    ELSE c.source
END source
from "TARGITO_207F62E1-FC2D-4D55-9C5C-B800034DCC62".PUBLIC.MAILING_ACTIONS m 
left join "TARGITO_207F62E1-FC2D-4D55-9C5C-B800034DCC62".ANALYSIS.CONTACTS_EXTENDED c
on m.contact_id = c.id
;
